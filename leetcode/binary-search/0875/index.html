0875. Koko Eating Bananas

If we set l < r and r = m

class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int h) {
        int l = 1;
        int r = *max_element(piles.begin(), piles.end());
        int m;
        long long hours;

        while(l <= r){
            m = l + (r - l) / 2;
            hours = 0;
            for(int& pile : piles){
                hours += ((pile / m) + (pile % m != 0));
            }

            if(hours == h){
                r = m - 1;
            } else if(hours > h){
                l = m + 1;
            } else {
                r = m - 1;
            }

        }

        return l;
    }
};
